const allPeople = await getCollection('people');
// Partners: top-level files (edgard.md, gustavo.md) — id has no '/'
const partners = allPeople.filter(p => !p.id.startsWith('advisors/'));
// Advisors: in advisors/ subdirectory
const advisors = allPeople.filter(p => p.id.startsWith('advisors/'));

// Photo URL helper — photos are in public/people/photos/ (copied from cmp-knowledge at build time)
// entry.data.photo = "photos/edgard-abreu.jpg" (relative from people/ dir)
// Target URL = "/people/photos/edgard-abreu.jpg"
const photoUrl = (entry: typeof allPeople[0]) => {
  if (!entry.data.photo) return '/people/photos/placeholder.jpg';
  const filename = entry.data.photo.split('/').pop() ?? '';
  return `/people/photos/${filename}`;
};
