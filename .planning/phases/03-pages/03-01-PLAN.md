// Auto-detect active nav link from current pathname (I18N-03, Phase 3)
const currentPath = Astro.url.pathname;
// Strip locale prefix so /pt/advisory matches href="/advisory"
const barePath = currentPath.replace(/^\/(pt|es)(\/|$)/, '/');
const isActive = (href: string): boolean => {
