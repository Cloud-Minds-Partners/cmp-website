    **LangSwitcher.astro — new file:**

    ```astro
    ---
    import { getRelativeLocaleUrl } from 'astro:i18n';

    const currentLocale = Astro.currentLocale ?? 'en';

    // Strip locale prefix from current path to get bare path for getRelativeLocaleUrl
    // Handles: /pt/team → /team, /es/ → /, /pt → /, /dev/components → /dev/components
    const rawPath = Astro.url.pathname
      .replace(/^\/(pt|es)(\/|$)/, '/')
      .replace(/\/$/, '') || '/';

    // getRelativeLocaleUrl expects path without leading slash
    const barePath = rawPath.replace(/^\//, '') || '';

    const locales = [
      { code: 'en', label: 'EN' },
      { code: 'pt', label: 'PT' },
      { code: 'es', label: 'ES' },
    ] as const;
    ---
    <div class="nav-lang" role="navigation" aria-label="Language selector">
      {locales.map((loc, i) => (
        <>
          <a
            href={getRelativeLocaleUrl(loc.code, barePath)}
            class={`lang-link${currentLocale === loc.code ? ' active' : ''}`}
            aria-current={currentLocale === loc.code ? 'page' : undefined}
          >{loc.label}</a>
          {i < locales.length - 1 && <span class="sep" aria-hidden="true">·</span>}
        </>
      ))}
    </div>
    <style>
      .nav-lang {
        display: inline-flex;
        align-items: center;
        font-family: var(--font-display);
        font-size: 13px;
    ```
